23  * some integer parameters, e.g., n_sites or n_fermions for the Hubbard example,
24  * or the number of sites per axis for a 3D Laplace discretization grid,
25  * should be "reasonably small" (even if the resulting matrix is "huge").
26  * By definition, an integer n is "reasonably small" if abs(n) <= SCAMACHUGEINT.
27  *
28  * SCAMACHUGEINT acts mainly as a failsafe against "irresponsible" parameter choices
29  * (e.g., when generating matrices in a loop).

32  * SCAMACHUGEINT should be much smaller than SCAMACIDXMAX.

34  * Note: The memory consumption for generator tables or workspace does not exceed
35  * a reasonable multiple of SCAMACHUGEINT (here, some MByte).
36  *
37  */

54 /* error codes */
55 /* we support up to 2^6 = 63 error codes + 1 "EOK"
56  * The error codes ENULL ... ECORRUPTED can be OR'ed with the position of the problematic input value,
57  * that is if the i-th parameter is problematic, with 1 <= i <= 31, functions return, e.g,
58  * SCAMAC_EINVALID | (i << SCAMAC_ESHIFT) or SCAMAC_EINVALID | (i << SCAMAC_ESHIFT) | SCAMAC_EINTERNAL
59  * The shift is given by SCAMAC_ESHIFT = 7
60  */
61 /* This definition must agree with the list of error names
62  in scamac_error_desc() in scamac_collection.c */
63 #define SCAMAC_ESHIFT 6
64 #define SCAMAC_EMASK ((1U << SCAMAC_ESHIFT) - 1)
65 typedef enum {
66 // +++ basic status
67 // routine completed successfully
68  SCAMAC_EOK=0,
69 // routine failed for an unspecified reason
70  SCAMAC_EFAIL,
71 // +++
72 // +++ problems with specific function parameters (on input)
73 // a null pointer was supplied in the wrong place
74  SCAMAC_ENULL,
75 // an input parameter, or the combination of all parameters, is invalid
76  SCAMAC_EINVALID,
77 // an (integer or double) input parameter is outside of the valid range (e.g, is negative when a positive value is expected)
78  SCAMAC_ERANGE,
79 // an object passed to a routine, probably constructed elsewhere, is corrupted. Can signal an internal error,
80 // or improper use of ScaMaC routines & algorithms.
81 // example: If a complex matrix is passed to a routine that expects a real matrix, EINVALID is returned.
82 // But if the valtype of the matrix is neither real nor complex, ECORRUPTED is returned, because that's not a valid matrix at all.
83  SCAMAC_ECORRUPTED,
84 // +++
85 // +++
86 // the requested computation is outside of the scope of the function (e.g., Lanczos for non-symmetric matrices)
87  SCAMAC_ESCOPE,
88 // the combination of all input parameters is invalid (but each individual parameter may be valid)
89  SCAMAC_EINPUT,
90 // +++
91 // +++ problems during execution or on return
92 // index/matrix dimension exceeds possible integer range (2^31 ~ 2E9 for int32, 2^63 ~ 9E18 for int64)
